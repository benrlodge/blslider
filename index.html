blSlider.js

Pass in a data attribute of images you want to use with bxslider.js

    How to use:

    	<link rel="stylesheet" href="css/jquery.bxslider.css">
	<script src="js/jquery.js"></script>
	<script src="js/blslider.js"></script>
	<script src="js/bxslider.js"></script>

	<ul class="bxslider slothSlider" data-images="img/sloth.jpeg, img/sloth.jpeg, 
	img/sloth.jpeg, img/sloth.jpeg, img/sloth.jpeg, img/sloth.jpeg, img/sloth.jpeg"></ul>

	<script>
	$(function(){
		
		$('.slothSlider').blSlider();

		$('.bxslider').bxSlider({
			minSlides: 4,
			maxSlides: 4,
			slideWidth: 300,
			slideMargin: 10
		});

	});
	</script>
